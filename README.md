   ```bash
   git clone https://github.com/yourusername/simon-karuga-portfolio.git
   cd simon-karuga-portfolio
   ```


   ```bash
   pip install -r requirements.txt
   ```

   ```bash
   python manage.py migrate
   ```

   ```bash
   python manage.py runserver
   ```
